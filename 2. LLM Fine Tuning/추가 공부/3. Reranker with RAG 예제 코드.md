```python
# ---------------------------------------------------------
# 1) Reranker 준비 (bge-reranker + LoRA)
# ---------------------------------------------------------
import torch
from transformers import AutoTokenizer, AutoModelForSequenceClassification

BASE_MODEL = "BAAI/bge-reranker-large"
LORA_ADAPTER = "./models/reranker_lora"  # 네가 학습한 LoRA 경로

tokenizer = AutoTokenizer.from_pretrained(BASE_MODEL)
model = AutoModelForSequenceClassification.from_pretrained(
    BASE_MODEL,
    device_map="auto"
)

# LoRA 로드
model.load_adapter(LORA_ADAPTER)
model.set_active_adapters("default")
model.eval()


# ---------------------------------------------------------
# 2) LangChain Custom Reranker 정의 (LCEL에서 사용)
# ---------------------------------------------------------
from langchain_core.runnables import Runnable, RunnableLambda

def reranker_fn(inputs):
    """
    inputs = {
        "query": "...",
        "docs": [Document, Document, ...]
    }
    """
    query = inputs["query"]
    docs = inputs["docs"]

    scores = []
    doc_texts = [d.page_content for d in docs]

    with torch.no_grad():
        for text in doc_texts:
            pairs = tokenizer(
                query,
                text,
                truncation=True,
                padding=True,
                return_tensors="pt"
            ).to(model.device)

            score = model(**pairs).logits.item()
            scores.append(score)

    # 점수 높은 순 정렬
    sorted_docs = [
        doc for _, doc in sorted(zip(scores, docs), key=lambda x: x[0], reverse=True)
    ]

    return sorted_docs


reranker = RunnableLambda(reranker_fn)


# ---------------------------------------------------------
# 3) Retriever 준비 (여기서는 간단히 Chroma 예시)
# ---------------------------------------------------------
from langchain_community.vectorstores import Chroma
from langchain.embeddings import HuggingFaceEmbeddings
from langchain_core.documents import Document

# 예시용 임베딩
embedding = HuggingFaceEmbeddings(model_name="sentence-transformers/all-MiniLM-L6-v2")

vectordb = Chroma(
    persist_directory="./chroma_db",
    embedding_function=embedding
)

retriever = vectordb.as_retriever(search_kwargs={"k": 8})


# ---------------------------------------------------------
# 4) LCEL Pipeline:  Retriever → Reranker
# ---------------------------------------------------------
# LCEL(Composable Graph)
pipeline = (
    {
        "query": lambda x: x["query"],
        "docs": lambda x: retriever.get_relevant_documents(x["query"])
    }
    | reranker
)


# ---------------------------------------------------------
# 5) 테스트 실행
# ---------------------------------------------------------
query = "LangGraph는 체크포인트를 어떻게 관리하나요?"
results = pipeline.invoke({"query": query})

print("\n=== 🔥 재순위된 최종 문서들 (점수 높은 순) ===\n")
for idx, doc in enumerate(results, start=1):
    print(f"[{idx}] {doc.page_content[:200]}")

```
