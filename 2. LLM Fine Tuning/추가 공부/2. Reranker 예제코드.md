# 📌 bge-reranker + Pairwise Loss + LoRA 파인튜닝 예제 코드

아래 코드는 다음 조합으로 실제 학습 가능한 형태입니다.

- 모델: `BAAI/bge-reranker-large`
- 학습 방법: Pairwise (positive vs negative)
- LoRA 적용
- PyTorch + HuggingFace Transformers + PEFT
- Trainer 없이 직접 학습 루프 구현

---

## 📂 데이터 형식 (train.jsonl 예시)

```json
{"query": "LangGraph에서 그래프 상태 저장 방식?", 
 "pos": "LangGraph는 checkpointing 기능으로 상태 저장을 지원합니다...", 
 "neg": "벡터 DB는 cosine similarity 기반 검색 방식을 사용합니다..."}
```

## 예제 코드 
```python
import torch
import json
from torch.utils.data import Dataset, DataLoader
from transformers import AutoTokenizer, AutoModelForSequenceClassification
from peft import LoraConfig, get_peft_model, prepare_model_for_kbit_training
from tqdm import tqdm


#########################################
# 1. Dataset 정의
#########################################

class PairwiseDataset(Dataset):
    def __init__(self, filepath, tokenizer, max_len=512):
        self.data = [json.loads(l) for l in open(filepath, 'r', encoding='utf-8')]
        self.tokenizer = tokenizer
        self.max_len = max_len

    def __len__(self):
        return len(self.data)

    def encode(self, query, passage):
        return self.tokenizer(
            query,
            passage,
            truncation=True,
            padding="max_length",
            max_length=self.max_len,
            return_tensors="pt",
        )

    def __getitem__(self, idx):
        item = self.data[idx]
        q, pos, neg = item["query"], item["pos"], item["neg"]

        pos_enc = self.encode(q, pos)
        neg_enc = self.encode(q, neg)

        return {
            "pos": pos_enc,
            "neg": neg_enc,
        }


#########################################
# 2. 모델 초기화 (bge-reranker + LoRA)
#########################################

model_name = "BAAI/bge-reranker-large"

tokenizer = AutoTokenizer.from_pretrained(model_name)
model = AutoModelForSequenceClassification.from_pretrained(
    model_name,
    torch_dtype=torch.bfloat16,
    device_map="auto"
)

# LoRA 설정
lora_config = LoraConfig(
    r=16,
    lora_alpha=32,
    lora_dropout=0.05,
    bias="none",
    target_modules=["query", "value", "key"],
)

model = prepare_model_for_kbit_training(model)
model = get_peft_model(model, lora_config)
model.print_trainable_parameters()


#########################################
# 3. DataLoader 준비
#########################################

dataset = PairwiseDataset("train.jsonl", tokenizer)
loader = DataLoader(dataset, batch_size=4, shuffle=True)

optimizer = torch.optim.AdamW(model.parameters(), lr=2e-5)


#########################################
# 4. Pairwise Loss 함수
#########################################

def pairwise_loss(score_pos, score_neg):
    diff = score_pos - score_neg
    return -torch.mean(torch.log(torch.sigmoid(diff)))


#########################################
# 5. 학습 루프
#########################################

epochs = 3

model.train()
for epoch in range(epochs):
    pbar = tqdm(loader, desc=f"Epoch {epoch+1}")

    for batch in pbar:
        optimizer.zero_grad()

        # positive 학습
        pos_input = {k: v.squeeze(1).to(model.device) for k, v in batch["pos"].items()}
        pos_out = model(**pos_input)
        score_pos = pos_out.logits  # shape: [B, 1]

        # negative 학습
        neg_input = {k: v.squeeze(1).to(model.device) for k, v in batch["neg"].items()}
        neg_out = model(**neg_input)
        score_neg = neg_out.logits

        # loss 계산
        loss = pairwise_loss(score_pos, score_neg)
        loss.backward()

        optimizer.step()

        pbar.set_postfix({"loss": loss.item()})


#########################################
# 6. LoRA Adapter 저장
#########################################

output_dir = "./bge-reranker-lora"
model.save_pretrained(output_dir)
tokenizer.save_pretrained(output_dir)

print(f"LoRA fine-tuned model saved to: {output_dir}")

```

## Inference 예제 코드

```python
from transformers import AutoTokenizer, AutoModelForSequenceClassification
import torch

model = AutoModelForSequenceClassification.from_pretrained(
    "./bge-reranker-lora",
    torch_dtype=torch.bfloat16,
    device_map="auto",
)
tokenizer = AutoTokenizer.from_pretrained("./bge-reranker-lora")

query = "LangGraph에서 checkpoint 저장 방식?"
doc = "LangGraph는 저장 가능한 graph state를 제공합니다..."

inputs = tokenizer(query, doc, return_tensors="pt").to(model.device)
score = model(**inputs).logits.item()

print("rerank score:", score)

```
